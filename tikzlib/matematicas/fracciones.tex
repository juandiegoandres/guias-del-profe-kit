%  tikzlib · matematicas/fracciones
%  Requiere: \usepackage{tikz}
%  \fraccionbarra{n}{k}   → barra de n partes iguales, k sombreadas
%  Color: el sombreado usa `accent` (por defecto verde si el host no lo define).
%  Preview: tikzlib/previews/fracciones.png
% =====================================================================
\providecolor{accent}{HTML}{1A8917}

\newcommand{\fraccionbarra}[2]{%
\begin{tikzpicture}[x=0.85cm, y=0.85cm, font=\sffamily\small]
  \foreach \i in {1,...,#1}{%
    \ifnum\i>#2 \fill[white] (\i-1,0) rectangle (\i,1);
    \else \fill[accent!30] (\i-1,0) rectangle (\i,1);\fi}
  \draw[line width=0.8pt] (0,0) rectangle (#1,1);
  \foreach \i in {1,...,#1}{\draw[line width=0.6pt] (\i,0) -- (\i,1);}
\end{tikzpicture}}
