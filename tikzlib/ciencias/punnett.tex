\newcommand{\punnettdos}[8]{%
\begin{tikzpicture}[font=\sffamily\large]
  \def\s{1.32}
  \foreach \i in {0,1,2}{%
    \draw[line width=0.9pt] (0,\i*\s) -- (2*\s,\i*\s);
    \draw[line width=0.9pt] (\i*\s,0) -- (\i*\s,2*\s);}
  \node at (0.5*\s,1.5*\s) {#5}; \node at (1.5*\s,1.5*\s) {#6};
  \node at (0.5*\s,0.5*\s) {#7}; \node at (1.5*\s,0.5*\s) {#8};
  \node at (0.5*\s,2*\s+0.42) {#1}; \node at (1.5*\s,2*\s+0.42) {#2};
  \node at (-0.45,1.5*\s) {#3}; \node at (-0.45,0.5*\s) {#4};
\end{tikzpicture}}

\newcommand{\punnettcuatro}[8]{%
\begin{tikzpicture}[font=\sffamily\normalsize]
  \def\s{1.25}
  \foreach \i in {0,1,2,3,4}{%
    \draw[line width=0.9pt] (0,\i*\s) -- (4*\s,\i*\s);
    \draw[line width=0.9pt] (\i*\s,0) -- (\i*\s,4*\s);}
  \node at (0.5*\s,4*\s+0.42) {#1}; \node at (1.5*\s,4*\s+0.42) {#2};
  \node at (2.5*\s,4*\s+0.42) {#3}; \node at (3.5*\s,4*\s+0.42) {#4};
  \node at (-0.5,3.5*\s) {#5}; \node at (-0.5,2.5*\s) {#6};
  \node at (-0.5,1.5*\s) {#7}; \node at (-0.5,0.5*\s) {#8};
\end{tikzpicture}}
